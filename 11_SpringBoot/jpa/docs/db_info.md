- MEMBER 테이블
  CREATE TABLE MEMBER (
  USER_ID VARCHAR(30) PRIMARY KEY,
  USER_PWD VARCHAR(100) NOT NULL,
  USER_NAME VARCHAR(15) NOT NULL,
  EMAIL VARCHAR(254),
  GENDER CHAR(1) CHECK (GENDER IN ('M', 'F')),
  AGE INT,
  PHONE VARCHAR(13),
  ADDRESS VARCHAR(100),
  ENROLL_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  MODIFY_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  STATUS CHAR(1) DEFAULT 'Y' NOT NULL CHECK (STATUS IN ('Y', 'N'))
  );
- PROFILE 테이블
  CREATE TABLE PROFILE (
  PROFILE_ID IDENTITY PRIMARY KEY,
  USER_ID VARCHAR(30) UNIQUE,
  PROFILE_IMAGE VARCHAR(100),
  INTRO VARCHAR(300),
  FOREIGN KEY (USER_ID) REFERENCES MEMBER(USER_ID) ON DELETE CASCADE
  );
- NOTICE 테이블
  CREATE TABLE NOTICE (
  NOTICE_NO IDENTITY PRIMARY KEY, -- AUTO_INCREMENT
  NOTICE_TITLE VARCHAR(30) NOT NULL,
  NOTICE_WRITER VARCHAR(30) NOT NULL,
  NOTICE_CONTENT VARCHAR(200) NOT NULL,
  CREATE_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (NOTICE_WRITER) REFERENCES MEMBER(USER_ID) ON DELETE CASCADE
  );
- BOARD 테이블
  CREATE TABLE BOARD (
  BOARD_NO IDENTITY PRIMARY KEY,
  BOARD_TITLE VARCHAR(100) NOT NULL,
  BOARD_WRITER VARCHAR(30) NOT NULL,
  BOARD_CONTENT CLOB NOT NULL, -- 큰 텍스트는 CLOB 사용
  ORIGIN_NAME VARCHAR(100),
  CHANGE_NAME VARCHAR(100),
  COUNT INT DEFAULT 0,
  CREATE_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  STATUS CHAR(1) DEFAULT 'Y' NOT NULL CHECK (STATUS IN ('Y', 'N')),
  FOREIGN KEY (BOARD_WRITER) REFERENCES MEMBER(USER_ID) ON DELETE CASCADE
  );
- REPLY 테이블
  CREATE TABLE REPLY (
  REPLY_NO IDENTITY PRIMARY KEY,
  REPLY_CONTENT VARCHAR(400) NOT NULL,
  REF_BNO INT NOT NULL,
  REPLY_WRITER VARCHAR(30) NOT NULL,
  CREATE_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
  STATUS CHAR(1) DEFAULT 'Y' NOT NULL CHECK (STATUS IN ('Y', 'N')),
  FOREIGN KEY (REF_BNO) REFERENCES BOARD(BOARD_NO) ON DELETE CASCADE,
  FOREIGN KEY (REPLY_WRITER) REFERENCES MEMBER(USER_ID) ON DELETE CASCADE
  );
- TAG 테이블
  CREATE TABLE TAG (
  TAG_ID IDENTITY PRIMARY KEY,
  TAG_NAME VARCHAR(30) NOT NULL UNIQUE
  );
- BOARD_TAG 테이블
  CREATE TABLE BOARD_TAG (
  BOARD_NO INT,
  TAG_ID INT,
  PRIMARY KEY (BOARD_NO, TAG_ID),
  FOREIGN KEY (BOARD_NO) REFERENCES BOARD(BOARD_NO) ON DELETE CASCADE,
  FOREIGN KEY (TAG_ID) REFERENCES TAG(TAG_ID) ON DELETE CASCADE
  );
